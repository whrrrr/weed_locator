#include <Arduino.h>
#include "pinout.h"

// 定义每秒多少个脉冲
const int stepDelayMicros = 1000; // 每一步延迟1000微秒 = 1kHz (约每秒1000步)
const int steps = 500; // 每次测试的步数

void setup() {
  Serial.begin(115200);

  // 配置引脚模式
  pinMode(X_STEP_PIN, OUTPUT);
  pinMode(X_DIR_PIN, OUTPUT);
  pinMode(X_ENABLE_PIN, OUTPUT);

  pinMode(Y_STEP_PIN, OUTPUT);
  pinMode(Y_DIR_PIN, OUTPUT);
  pinMode(Y_ENABLE_PIN, OUTPUT);

  pinMode(Z_STEP_PIN, OUTPUT);
  pinMode(Z_DIR_PIN, OUTPUT);
  pinMode(Z_ENABLE_PIN, OUTPUT);

  // 使能电机
  digitalWrite(X_ENABLE_PIN, LOW);
  digitalWrite(Y_ENABLE_PIN, LOW);
  digitalWrite(Z_ENABLE_PIN, LOW);

  Serial.println("3轴测试启动");
}

void stepMotor(int stepPin, int dirPin, bool dir, int numSteps) {
  digitalWrite(dirPin, dir);
  for (int i = 0; i < numSteps; i++) {
    digitalWrite(stepPin, HIGH);
    delayMicroseconds(stepDelayMicros);
    digitalWrite(stepPin, LOW);
    delayMicroseconds(stepDelayMicros);
  }
}

void loop() {
  Serial.println("X轴正转");
  stepMotor(X_STEP_PIN, X_DIR_PIN, true, steps);
  delay(500);
  Serial.println("X轴反转");
  stepMotor(X_STEP_PIN, X_DIR_PIN, false, steps);
  delay(500);

  Serial.println("Y轴正转");
  stepMotor(Y_STEP_PIN, Y_DIR_PIN, true, steps);
  delay(500);
  Serial.println("Y轴反转");
  stepMotor(Y_STEP_PIN, Y_DIR_PIN, false, steps);
  delay(500);

  Serial.println("Z轴正转");
  stepMotor(Z_STEP_PIN, Z_DIR_PIN, true, steps);
  delay(500);
  Serial.println("Z轴反转");
  stepMotor(Z_STEP_PIN, Z_DIR_PIN, false, steps);
  delay(500);
}
